Sample Input File for FEA Analysis

Deprecated documentation: the new version is in the OSUPDocs wiki

The following XML file can be input into NairnFEA to do calculations of an end-loaded cantilever beam. Certain parameters of the problem can be set by using the XML ENTITY definitions in the DOCTYPE element at the start of the file. These settings can select the beam material modulus, its dimensions, and the total end load.

<?xml version='1.0'?>
<!DOCTYPE JANFEAInput SYSTEM 'NairnFEA.dtd'
[
 <!ENTITY modulus '2300'>
 <!ENTITY length '50'>
 <!ENTITY depth '10'>
 <!ENTITY thickness '10'>
 <!ENTITY load '500'>
]>
<JANFEAInput version='3'>

  <Header>
    <Description>
  A cantilever beam that is clamped at its base and
  loaded on its free end with a single point load.
    </Description>
    <Analysis>1</Analysis>
  </Header>

  <Mesh>
    <Keypoints>
      <pt x='0' y='0' id='lowerleft'/>
      <pt x='0' y='&depth;' id='upperleft'/>
      <pt x='&length;' y='0' id='lowerright'/>
      <pt x='&length;' y='&depth;' id='upperright'/>
    </Keypoints>

    <Path id='bottom' intervals='6'>
      <keypt id='lowerleft'/>
      <keypt id='lowerright'/>
    </Path>
    <Path id='right' intervals='3'>
      <keypt id='lowerright'/>
      <keypt id='upperright'/>
    </Path>
    <Path id='top' intervals='6'>
      <keypt id='upperright'/>
      <keypt id='upperleft'/>
    </Path>
    <Path id='left' intervals='3'>
      <keypt id='upperleft'/>
      <keypt id='lowerleft'/>
    </Path>

    <Area mat='1' thick='&thickness;' type='3'>
      <path id='bottom'/>
      <path id='right'/>
      <path id='top'/>
      <path id='left'/>
    </Area>
  </Mesh>

  <Material Type='1' Name='Polymer'>
    <E>&modulus;</E>
    <nu>0.33</nu>
    <alpha>100</alpha>
  </Material>

  <GridBCs>
    <BCLine x1='0' y1='0' x2='0' y2='&depth;' tolerance='.1'>
      <DisBC dof='1'/>
      <DisBC dof='2'/>
    </BCLine>

    <BCPt keypt='upperright'>
      <LoadBC dof='2' load='&load;'/>
    </BCPt>
  </GridBCs>
</JANFEAInput>
